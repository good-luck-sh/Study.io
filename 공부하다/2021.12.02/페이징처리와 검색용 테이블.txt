페이징처리와 검색용 테이블과 시퀀스

CREATE TABLE sample_search_boards (
	BOARD_ID NUMBER(8) PRIMARY KEY,
	BOARD_TITLE VARCHAR2(255) NOT NULL,
	BOARD_WRITER VARCHAR2(255) NOT NULL,
	BOARD_CONTENT VARCHAR2(100) NOT NULL,
	BOARD_VIEW_COUNT NUMBER(8) DEFAULT 0,
	BOARD_DELETED CHAR(1) DEFAULT 'N',
	BOARD_UPDATED_DATE DATE,
	BOARD_CREATED_DATE DATE DEFAULT SYSDATE
);

CREATE SEQUENCE sample_search_boards_seq;
